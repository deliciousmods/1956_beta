BHU_bhutan_diplomacy = {
	BHU_denounce_the_treaty_of_punakha = {

		icon = generic_break_treaty

		allowed = {
			tag = BHU
		}

		available = {
			has_completed_focus = BHU_new_ideas
			has_war = no
			or = {
				is_subject_of = ENG
				is_puppet = no
			}
			OR = {
				fascism > 0.5
				communism > 0.5  				
				democratic > 0.5  
			}	
		}

		fire_only_once = yes

		ai_will_do = {
			factor = 10
		}
		
		days_remove = 10
		cost = 100
		
		visible = {
			tag = BHU
		}
		remove_effect = {
			end_puppet = ENG
			leave_faction = yes
			set_country_flag = denounced_punakha_treaty
			if = {
				limit = {
					has_idea = BHU_treaty_with_britain
				}
				remove_ideas = BHU_treaty_with_britain
			}
			if = {
				limit = {
					is_puppet = no
				}
				add_political_power = 110 #just to be sure that it's always removable
			}
		}
	}
	BHU_improve_relations = {

		icon = generic_political_discourse

		allowed = {
			tag = BHU
		}
		
		available = {
			tag = BHU
			OR = {
				has_completed_focus = BHU_the_quiet_strength
				has_completed_focus = BHU_aristocratic_regime
				has_completed_focus = BHU_overthrow_the_feudals
				has_completed_focus = BHU_spread_aggressive_attitude			
			}
			FROM = {
				is_puppet = no
				exists = yes
			}
			OR = {				
				BHU = {
					has_government = FROM
				}
				AND = {
					BHU = {
						OR = {
							has_government = neutrality
							has_government = democratic							
						}
						
					}
					FROM = {
						OR = {
							has_government = neutrality
							has_government = democratic							
						}
					}
				}				
			}		
		}

		target_trigger = {
			FROM = {
				is_neighbor_of = ROOT
				has_opinion = {
					target = ROOT
					value > 19
				}	
			}
		}		
		
		cost = 50
		fire_only_once = yes

		ai_will_do = {
			factor = 10
		}
		
		days_remove = 30
		
		visible = {
			FROM = {
				is_puppet = no
				exists = yes
			}
		}
		remove_effect = {
			set_country_flag = BHU_diplomacy_decision
			add_opinion_modifier = {
				target = FROM
				modifier = large_increase
			}
		}
	}
