				}
				add_popularity = {
					ideology = fascism
					popularity = 0.04
				}
				add_popularity = {
					ideology = democratic
					popularity = 0.10
				}
				add_popularity = {
					ideology = neutrality
					popularity = 0.30
				}
				random_other_country = {
					limit = {
						original_tag = NOR
						has_government = democratic
					}
					set_equipment_fraction = 0.5
					set_stability = 0.85
					set_war_support = 0.85
					load_oob = "NOR_Norwegian_Civil_War_democratic"
					add_popularity = {
						ideology = communism
						popularity = 0.04
					}
					add_popularity = {
						ideology = fascism
						popularity = 0.04
					}
					add_popularity = {
						ideology = neutrality
						popularity = 0.15
					}
					add_popularity = {
						ideology = democratic
						popularity = 0.10
					}
					unlock_national_focus = NOR_affirm_constitutional_monarchy
				}
			}
			
			set_war_support = 0.25
			
			hidden_effect = {
				if = {
					limit = {
						has_country_flag = NOR_ensured_loyalty_among_commanders
					}
					random_other_country = {
						limit = {
							original_tag = NOR
							NOT = { has_government = neutrality }
						}
						every_unit_leader = {
							set_nationality = NOR
						}
					}
					clr_country_flag = NOR_ensured_loyalty_among_commanders
				}
				if = {
					limit = {
						OR = {
							has_country_flag = NOR_opened_political_discourse_monarchism
							has_country_flag = NOR_preparation_for_monarchist_civil_war
							#has_country_flag = NOR_civil_support_for_monarchist_civil_war
							has_country_flag = NOR_passed_first_civil_war_opportunity
						}
					}
					clr_country_flag = NOR_opened_political_discourse_monarchism
					clr_country_flag = NOR_preparation_for_monarchist_civil_war
					#clr_country_flag = NOR_civil_support_for_monarchist_civil_war
					clr_country_flag = NOR_passed_first_civil_war_opportunity
				}
				set_country_flag = NOR_has_monarchist_civil_war
			}
		}

			factor = 1
