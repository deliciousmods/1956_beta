		fire_only_once = yes

		complete_effect = {

		}

		ai_will_do = {

		}

	ARG_the_end_of_justo_political_life = {

		allowed = {
			original_tag = ARG
		}

		priority = 1
		icon = generic_political_discourse

		visible = {
			date > 1937.10.1
			has_country_leader = {
				character = ARG_agustin_pedro_justo
				ruling_only = yes
			}			
		}
		activation = {
			date > 1937.10.1
			has_country_leader = {
				character = ARG_agustin_pedro_justo
				ruling_only = yes
			}		
		}

		available = {
			OR = {
				date > 1938.2.19
				NOT = {
					has_government = neutrality
				}
			}
			
		}
		cancel_if_not_visible = no

		days_mission_timeout = 153 #roughly the date of the new presidency, to adapt later

		fire_only_once = yes


		#Fail
		timeout_effect = {
			if = {
				limit = {
					NOT = {
						has_government = neutrality
					}
				}
				retire_ideology_leader = neutrality
				add_popularity = {
					ideology = neutrality
					popularity = -0.05
				}				
			}
			else = {
				retire_ideology_leader = neutrality
				add_stability = 0.01				
			}
		}
		#Success
		complete_effect = {
			retire_ideology_leader = neutrality
			add_stability = 0.01
		}			
		ai_will_do = {
			factor = 50
		}
	}
