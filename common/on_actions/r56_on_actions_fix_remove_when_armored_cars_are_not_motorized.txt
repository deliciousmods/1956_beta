#granted with authorization by a friendly redditor u/blitzgunner at the time I am writing this the AI is told to build motorized equipment to help supplies but can't tell the difference between trucks and armored cars, so it builds lots and lots of armored cars.
on_actions = {
	on_monthly	= {
		effect = {
			if = {
			limit = {
						is_ai = yes
					}
				add_equipment_to_stockpile = {
				type = motorized_equipment 
				amount = num_equipment@armored_car_equipment 
				}
			
				set_temp_variable = { armored_cars_count = 0 }
				subtract_from_temp_variable = {armored_cars_count = num_equipment@armored_car_equipment }
				
				add_equipment_to_stockpile = {
				type = armored_car_equipment 
				amount = armored_cars_count
				}
			}
		}
	}
}
