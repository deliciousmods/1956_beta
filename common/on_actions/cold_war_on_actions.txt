		effect = {
			if = {
				limit = {
					original_tag = HUN
					date < 1946.01.01
					has_start_date > 1945.01.01
				}
				country_event = cold_war.1
			}
			else_if = {
				limit = {
					original_tag = HUN
					date < 1948.01.01		
					has_start_date > 1945.01.01		
				}
				set_politics = {

					elections_allowed = no
				}
			}

				limit = {
					original_tag = GRE
					date < 1947.01.01
					has_start_date > 1945.01.01
