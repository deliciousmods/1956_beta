		has_dlc = "La Resistance"		
	}
	on_enable = {
		custom_effect_tooltip = resistance_60_effect_tooltip
		hidden_effect = {
			# avoid effects that can effect state controllers here
			#country_event = { id = occupied_countries.2 hours = 1 }

				}
				FROM = {
					country_event = {
						id = r56_spying_event.11
						hours = 1
					}
					
				}				
			}
		}
	}
}
