						has_tech = centimetric_radar
					}
				} 
			}
			FROM = { 
				OR = {
					has_tech = improved_decimetric_radar 
					has_tech = centimetric_radar
				}
			}
		}
		else_if = {
			limit = {
				ROOT = { 
					has_tech = centimetric_radar

						has_tech = improved_centimetric_radar
					}
				}

					has_tech = centimetric_radar
