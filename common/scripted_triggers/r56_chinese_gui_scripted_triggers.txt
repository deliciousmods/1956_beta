chi_is_blue_warlord = {
	custom_trigger_tooltip = {
		tooltip = "Is Kuomintang-Aligned"
		CHI = { r56_is_literally_kuomintang = yes }
		CONTROLLER = {
			NOT = {
				has_war_with = CHI
			}
			OR = {
				tag = CHI
				is_subject_of = CHI
				AND = {
					tag = XSM
					has_country_leader = {
						character = XSM_ma_bufang
						ruling_only = yes
					}
					OR = {
						has_government = democratic
						has_government = neutrality
					}
				}
				AND = {
					tag = YUN
					has_country_leader = {
						character = YUN_long_yun
						ruling_only = yes
					}
					OR = {
						has_government = democratic
						has_government = neutrality
					}
					# TODO back to neutral after sino-jap war
				}
				AND = {
					tag = SHX
					has_country_leader = {
						character = SHX_yan_xishan
						ruling_only = yes
					}
					OR = {
						has_government = democratic
						has_government = neutrality
					}
				}
				AND = {
					tag = TNG
					has_country_leader = {
						character = TNG_ma_hushan
						ruling_only = yes
					}
					OR = {
						has_government = democratic
						has_government = neutrality
					}
				}
				AND = {
					tag = KOR
					is_in_faction_with = CHI
				}
				# Guangxi opposed to Chiang before war
				AND = {
					tag = GXC
					has_war_with = JAP
					CHI = { has_war_with = JAP }
					OR = {
						has_government = democratic
						has_government = neutrality
					}
				}
				#has_country_flag = kmt_aligned
			}
		}
	}
