# Don't lend lease to ETh
ETH_lon_dont_support_ethiopia_with_weapons = {

	allowed = { has_dlc = "By Blood Alone" }
	enable = {
		is_league_of_nations_member = yes 
		NOT = {
			tag = ETH
		}
	}

	abort = {
		OR = {

		}
	}

		type = send_lend_lease_desire

		value = -200
	}
}
