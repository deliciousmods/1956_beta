HON_fascist_boy_perfers_GER_over_JAP = {
	enable = {
		original_tag = HON
		HON = {
			has_government = fascism
		}
		country_exists = HON
		country_exists = GER
		GER = {
			has_government = fascism
			is_faction_leader = yes
			surrender_progress < 0.05
		}
		JAP = {
			has_government = fascism
			is_faction_leader = yes
		}
	}
	
	abort_when_not_enabled = yes
	
	ai_strategy = {
		type = befriend
		id = "GER"	
		value = 25
	}
	
	ai_strategy = {
		type = befriend
		id = "JAP"	
		value = -100
	}

}
