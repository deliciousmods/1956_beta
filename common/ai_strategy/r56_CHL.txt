CHL_be_nice_to_germany_early = {

		original_tag = CHL
		date < 1937.3.1
		threat < 0.16
		GER = {
			has_war = no
		}
		country_exists = GER
		GER = {
			OR = {
				NOT = {
					has_government = democratic
					has_government = communism
				}
			}
		}
		CHL = {
			OR = {
				has_government = democratic
				has_government = neutrality
			}
		}
	}

	abort_when_not_enabled = yes
	

	ai_strategy = {
		type = befriend
		id = "GER"	
		value = 15
	}

}


CHL_play_nice_with_ARG = {
	enable = {
		original_tag = CHL
		CHL = {
			OR = {
				NOT = { has_completed_focus = CHL_the_spread_of_fascism }
				NOT = { has_completed_focus = CHL_communist_argentina_war }
			}
		}
		ARG = {
			NOT = { has_completed_focus = ARG_argentinas_destiny }
			
		}
		country_exists = ARG
		ARG = {
			OR = {
				NOT = {
					has_government = communism
					has_government = fascism
				}
			}
		}
		CHL = {
			OR = {
				has_government = democratic
				has_government = neutrality
			}
		}
	}
	abort = {
		OR = {
			ARG = {
				has_government = fascism
			}
			NOT = {
				country_exists = ARG
			}
			CHL = {
				has_war_with = ARG
			}
		}
	}

	ai_strategy = {
		type = befriend
		id = "ARG"	
		value = 20
	}

	ai_strategy = {
		type = support
		id = "ARG"
		value = 1
	}

}

CHL_nice_demo_boy_stays_home_unless_something_goes_wrong = {
	enable = {
		original_tag = CHL
		CHL = {
			AND = {
				exists = yes
				has_government = democratic
				has_completed_focus = CHL_join_allies
				OR = {
					is_in_faction_with = ENG
					is_in_faction_with = USA
				}
				has_war_with = GER
			}
		}
		OR = {
			USA = {
				exists = yes
				has_government = democratic
				is_in_faction_with = CHL
				surrender_progress < 0.02
				has_war_with = GER
			}
			ENG = {
				exists = yes
				has_government = democratic
				is_in_faction_with = CHL
				surrender_progress < 0.02
				has_war_with = GER
			}
		}
	}
	
	abort_when_not_enabled = yes

	ai_strategy = {
		type = dont_defend_ally_borders
		id = "USA"
		value = 100
	}

	ai_strategy = {
		type = dont_defend_ally_borders
		id = "ENG"
		value = 100
	}

}
