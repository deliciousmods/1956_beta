CHL_play_nice_with_ARG = {
	allowed = {
		original_tag = CHL
	}

	enable = {
		#original_tag = CHL
		ROOT = {
			exists = yes
			OR = {
				NOT = { has_completed_focus = CHL_the_spread_of_fascism }
				NOT = { has_completed_focus = CHL_communist_argentina_war }
			}
		}
		ARG = {
			NOT = { has_completed_focus = ARG_argentinas_destiny }
			
		}
		country_exists = ARG
		ARG = {
			OR = {
				NOT = {
					has_government = communism
					has_government = fascism
				}
			}
		}
		CHL = {
			OR = {
				has_government = democratic
				has_government = neutrality
			}
		}
	}
	abort = {
		OR = {
			ARG = {
				has_government = fascism
			}
			NOT = {
				country_exists = ARG
			}
			CHL = {
				has_war_with = ARG
			}
		}
	}

	ai_strategy = {
		type = befriend
		id = "ARG"	
		value = 10
	}

}

		}
		country_exists = SPR
		SPR = {
			OR = {
				has_government = democratic
				has_government = communism
			}
		}
		CHL = {
			has_government = communism
		}
		SPR = { has_civil_war = yes }

			has_global_flag = scw_is_ai_driven

				is_ai = no
			}

			date > 1939.1.1
