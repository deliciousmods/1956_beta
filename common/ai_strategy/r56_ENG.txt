ENG_less_mean_to_SOV_early = {
	enable = {
		tag = ENG
		ENG = {
			has_government = democratic
			NOT = { has_war_with = SOV }
		}
		country_exists = SOV
		SOV = {
			is_ai = yes
			has_government = communism
		}
		any_country = {
			is_major = yes
			NOT = {
				tag = SOV
			}
			has_war_with = ENG		
		}		
	}
	abort = {
		OR = {
			NOT = { country_exists = SOV }
			date > 1941.1.1
		}
	}

	ai_strategy = {
		type = antagonize
		id = "SOV"			
		value = -25
	}

	ai_strategy = {
		type = contain
		id = "SOV"			
		value = -25
	}	

	ai_strategy = {
		type = befriend
		id = "SOV"			
		value = 10
	}
}

	enable = {
		original_tag = ENG
		has_civil_war = no
		OR = {
			has_war_with = FRA
			AND = {
				FRA = {
					has_capitulated = yes
				}
				has_war = yes
			}
		}
		any_country = {
			OR = {
				capital_scope = {
					is_on_continent = europe
				}
				AND = {
					is_major = yes
					is_ai = no
				}				
			}
			OR = {
				has_government = communism
				has_government = fascism
				has_war_with = ROOT
			}
			NOT = {
				OR = {
					is_in_faction_with = ROOT
					is_puppet = yes
				}
				
			}
			num_divisions > 20
		}
	}
	
	abort = {
		
	}
	
	ai_strategy = {
		type = put_unit_buffers
		ratio = 0.2 
		states = { 
			136
 			126
 			127
 			128
 			129
 			338
 			123
 			122 
 		}	
	}
