EAI_JAP_stalemate_vs_CHI = {
	allowed = { tag = JAP }
	enable = {
		has_war_with = CHI
		CHI = { is_ai = yes }
		check_variable = { global.EAI_sino_japanese_war = 2 } # stalemate

		# Create some offensives every once in a while to make the front seem active
		set_temp_variable = { day = global.EAI_days_elapsed }
		modulo_temp_variable = { day = 30 }
		OR = {
			check_variable = { day < 26 }
			CHI = { surrender_progress > 0.6 }
			check_variable = { EAI_fielded_eq_ratio < 0.9 }
			EAI_reserve_manpower_less_than_25p = yes
		}

		### Conditions
		
		OR = {

			# Stop attacks when JAP has pushed far enough
			AND = { 

				CHI = { surrender_progress > 0.5 }

				# Keep pushing if JAP's puppets are losing ground
				OR = {

					NOT = {
						MAN = { is_in_faction_with = JAP surrender_progress > 0 }
						MEN = { is_in_faction_with = JAP surrender_progress > 0 }
					}

					CHI = { surrender_progress > 0.75 }
				}
			}

			# Stop attacks if JAP is losing the war
			AND = {

				JAP = { surrender_progress > 0.2 }
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control
		priority = 10
		tag = CHI
		execute_order = no
	}
}

# if china has fallen and we are going for USA we will end up at war with allies suddenly while maybe bordering them,
# but thats hard to guess for AI considering all the steps. So we warn it so it can prep vs the allies
jap_early_part_beware_allies = {
	allowed = {
		original_tag = JAP
	}
	enable = {
		USA = { is_in_faction = no }
		focus_progress = {
			focus = JAP_secure_the_philippines
			progress > 0
		}
		PHI = {	is_subject_of = USA }
		NOT = { has_war_together_with = ENG }
	}

	abort = {
		OR = {
			USA = { is_in_faction = yes } # they know where to stand so we can go back to regular logic
			has_war_together_with = ENG
		}
	}

	ai_strategy = {
		type = prepare_for_war
		id = "ENG"
		value = 100
	}
	ai_strategy = {
		type = prepare_for_war
		id = "FRA"
		value = 100
	}
	ai_strategy = {
		type = prepare_for_war
		id = "PHI"
		value = 100
	}
	ai_strategy = {
		type = prepare_for_war
		id = "USA"
		value = 100
	}
	ai_strategy = {
		type = prepare_for_war
		id = "RAJ"
		value = 100
	}
}
