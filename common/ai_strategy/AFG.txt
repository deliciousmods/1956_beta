AFG_save_pp_for_momand = {
	allowed = {
		original_tag = AFG
	}
	enable = {
		AFG = {
			has_completed_focus = AFG_first_loya_jirga
			has_active_mission = AFG_second_loya_jirga_mission
			has_country_flag = AFG_win_loya_jirga
			NOT = {
				AFG_mohammad_gul_khan_momand = {
					is_political_advisor = yes
				}
				power_balance_value = {
					id = AFG_power_balance
					value > 0.45
				}
			}
		}
	}

	abort = {
		AFG = {
			OR = {
				AFG_mohammad_gul_khan_momand = {
					is_political_advisor = yes
				}
				NOT = {
					has_active_mission = AFG_second_loya_jirga_mission
				}
			}
		}
	}

	ai_strategy = {
		type = pp_spend_amount
		id = idea
		value = 150
	}

	ai_strategy = {
		type = pp_spend_priority
		id = idea
		value = 150
	}
}

AFG_send_vols_to_TNG = {
	allowed = {
		original_tag = AFG
	}
	enable = {
		TNG = {
			has_war_with = SIK
		}
		AFG = {
			has_idea = AFG_volunteers_in_tunganistan
		}
	}

	abort = {
		OR = {
			TNG = {
				exists = no
			}
			TNG = {
				NOT = {
					has_war_with = SIK
				}
			}
			AFG = {
				NOT = {
					has_idea = AFG_volunteers_in_tunganistan
				}
			}
		}
	}

	ai_strategy = {
		type = send_volunteers_desire
		id = TNG
		value = 25
	}

	ai_strategy = {
		type = support
		id = TNG
		value = 25
	}

	ai_strategy = {
		type = theatre_distribution_demand_increase
		value = 75
		id = 619
	}
}
