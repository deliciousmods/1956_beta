	allowed = {
		OR = {
			tag = SPA
			tag = SPB
			tag = SPC
			tag = SPD
			original_tag = SPR
		}
	}
	enable = {
		has_country_flag = EAI_SCW_limiter
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control
		priority = 10
		state_trigger = {
			is_in_array = { global.EAI_spanish_civil_war_states_array = THIS.id }
			OR = {
				is_controlled_by = SPA
				is_controlled_by = SPB
				is_controlled_by = SPC
				is_controlled_by = SPD
				is_controlled_by = SPR
			}
		}

		execution_type = careful
		execute_order = no
	}
}

# Prevent Nationalist Spain to join any faction on historical mode
nationalist_spain_allies_nobody_on_historical = {
	allowed = {
		tag = SPA
		has_dlc = "La Resistance"
	}
	enable = {
		has_global_flag = scw_over
		OR = {
			AND = {
				is_historical_focus_on = yes
				has_game_rule = {
					rule = SPR_ai_behavior
					option = DEFAULT
				}
			}
			has_country_flag = SPR_AI_RANDOM_NATIONALIST_HISTORICAL
			has_game_rule = {
				rule = SPR_ai_behavior
				option = NATIONALIST_HISTORICAL
			}
		}
	}
	abort = {
		OR = {
			SPA = { has_war = yes }
			SPA = { is_in_faction = yes }
			GER = { NOT = { has_government = fascism } }
			ENG = { NOT = { has_government = democratic } }
		}
	}

	ai_strategy = {
		type = alliance
		id = GER
		value = -200
	}

	ai_strategy = {
		type = alliance
		id = ENG
		value = -200
	}
}
