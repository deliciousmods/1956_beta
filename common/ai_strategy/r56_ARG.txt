ARG_be_nice_to_germany_early = {
	
	allowed = {
		original_tag = ARG
	}
	enable = {
		#original_tag = ARG
		date < 1937.3.1
		threat < 0.16
		GER = {
			has_war = no
		}
		country_exists = GER
		GER = {
			OR = {
				NOT = {
					has_government = democratic
					has_government = communism
				}
			}
		}
		ARG = {
			OR = {
				has_government = democratic
				has_government = neutrality
			}
		}
	}

	abort_when_not_enabled = yes
	

	ai_strategy = {
		type = befriend
		id = "GER"	
		value = 15
	}

}


ARG_play_nice_with_CHL = {

	allowed = {
		original_tag = ARG
	}

	enable = {
		#original_tag = ARG
		CHL = {
			OR = {
				NOT = { has_completed_focus = CHL_the_spread_of_fascism }
				NOT = { has_completed_focus = CHL_communist_argentina_war }
			}
		}
		ARG = {
			NOT = { has_completed_focus = ARG_argentinas_destiny }
			
		}
		country_exists = CHL
		CHL = {
			OR = {
				NOT = {
					has_government = communism
					has_government = fascism
				}
			}
		}
		ARG = {
			OR = {
				has_government = democratic
				has_government = neutrality
			}
		}
	}
	abort = {
		OR = {
			CHL = {
				has_government = fascism
			}
			NOT = {
				country_exists = CHL
			}
			ARG = {
				has_war_with = CHL
			}
		}
	}

	ai_strategy = {
		type = befriend
		id = "CHL"	
		value = 20
	}

	ai_strategy = {
		type = support
		id = "CHL"
		value = 10
	}

}
