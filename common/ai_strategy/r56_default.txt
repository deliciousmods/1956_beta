	enable = {

		num_of_military_factories > 69
	}
	abort = {
		num_of_military_factories < 30
	}

		type = role_ratio

		value = 8
	}
}

r56_help_jap_deal_with_partisans_and_fronts = {

	enable = {
		NOT = { original_tag = JAP }
		JAP = {
			is_major = yes
			is_faction_leader = yes
			exists = yes
			has_offensive_war = yes
			NOT = {
				OR = {
					has_war_with = SOV
					has_war_with = MON
				}
			}
		}
		ROOT = {
			is_subject_of = JAP
			surrender_progress < 0.02
		}
	}

	ai_strategy = {
		type = spare_unit_factor		
		value = 0
	}

	ai_strategy = {
		type = support
		id = JAP
		value = 250
	}
}

			AND = {
				has_government = communism

				is_ai = yes
				is_major = yes
			}
		}
	}
	
	abort_when_not_enabled = yes

	ai_strategy = {
		type = befriend
		id = SOV
		value = -10

	ai_strategy = {
		type = alliance
