#Below is a simple 'Bait and Switch' custom modifier blueprint using country flags and custom modifiers. 
#This is an example of using country flags that are attached to basegame files to create 'fake' basegame-like effects/modifiers. Adopt and change to your own use case. - Spicyalfredo

#CUSTOM 'X Can Switch Land Doctrine' GAMERULE EFFECT DOCUMENTION#
#To use this custom gamerule follow these steps:
# - This effect is already on the land_doctrine tech file and works for all countries when the country flag is set. Look at that file if you're making a new effect from nothing to see how it's done. 
# - You need to set both set_country_flag's seen below in your history file/desired effect using 'r56_set_the_rule_cannot_switch_land_doctrines = yes' below. 
# - Add 'custom_modifier_tooltip = r56_can_switch_land_doctrines_game_rule_modifier_tt' to your Idea/Dynamic Modifier that you want to tie to preventing the land doctrine switches.
# - To clear the rule simply use the script r56_set_the_rule_can_switch_land_doctrines = yes' 
# - Add 'custom_effect_tooltip = r56_cannot_switch_doctrine_gamerule_set_rule_yes_tt' effect after the clr_country_flag = r56_cannot_switch_doctrine_gamerule hidden effect to complete this custom modifier illusion  
# - Swap the idea Idea/Dynamic Modifier that has the 'custom_modifier_tooltip = r56_can_switch_land_doctrines_game_rule_modifier_tt' to one that doesn't in the same effect. 
# - To use this as an active effect use 'r56_set_the_rule_cannot_switch_land_doctrines = yes.' It's suggested that you also attach 'custom_effect_tooltip = r56_cannot_switch_doctrine_gamerule_set_rule_no_tt' to desired idea/content.
# - To remove simply use 'r56_set_the_rule_cannot_switch_land_doctrine = yes'
# - Working example of this in action is in the Irish Focus tree focus "IRE_a_doctrinal_change_in_course" 

#Adds the rule
r56_set_the_rule_cannot_switch_land_doctrines = { #It's sugggested you tie this effect an idea/content that contains the custom modifier 'r56_can_switch_land_doctrines_game_rule_modifier_tt' to highlight this effect. If not 
	custom_effect_tooltip = r56_cannot_switch_doctrine_gamerule_set_rule_no_tt #The 'effect'
	hidden_effect = {
		ROOT = {
			set_country_flag = r56_root_had_cannot_switch_doctrine_gamerule_at_start #This is tied to the latter and is used for tracking and a cleaner tooltip on the land doctrine tech file. Both are needed for the 'effect' to appear polished.
			set_country_flag = r56_cannot_switch_doctrine_gamerule #Sets custom r56 cannot switch Land Doctrine 'gamerule'
		}
	}
}

#Removes the rule
r56_set_the_rule_can_switch_land_doctrines = { #You'll still need to swap the idea/content that contained the custom modifier 'r56_can_switch_land_doctrines_game_rule_modifier_tt' to one that doesn't. 
	custom_effect_tooltip = r56_cannot_switch_doctrine_gamerule_set_rule_yes_tt #The 'effect'
	hidden_effect = {
		ROOT = {
			clr_country_flag = r56_cannot_switch_doctrine_gamerule #only clear this flag. This is on purpose. 
		}
	}
}
