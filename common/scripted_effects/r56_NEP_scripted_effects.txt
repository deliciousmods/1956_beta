	# level 4 not relevant. Cannot go above.

			if = {
				limit = {
					has_idea = NEP_illiteracy_5
				}

				}
				swap_ideas = {

					add_idea = NEP_illiteracy_5
				}
			}
			else_if = {
				limit = {
					has_idea = NEP_illiteracy_3
				}
				swap_ideas = {
					remove_idea = NEP_illiteracy_3
					add_idea = NEP_illiteracy_4
				}
			}
			else_if = {
				limit = {
					has_idea = NEP_illiteracy_2
				}
				swap_ideas = {
					remove_idea = NEP_illiteracy_2
					add_idea = NEP_illiteracy_3
				}
			}
			else_if = {
				limit = {
					has_idea = NEP_illiteracy_1
				}
				swap_ideas = {
					remove_idea = NEP_illiteracy_1
					add_idea = NEP_illiteracy_2
				}
			}
			else_if = {
				limit = {
					has_idea = NEP_illiteracy
				}
				swap_ideas = {
