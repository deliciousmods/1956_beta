# in on_daily, ran for every country
update_majors_array = {
	if = {
		limit = { 
			is_major = yes
			NOT = {
				is_in_array = {
					array = global.majors
					value = THIS
				}
			}
		}
		add_to_array = {
			global.majors = THIS
		}
	}
	else_if = {
		limit = {
			is_major = no
			is_in_array = {
				array = global.majors
				value = THIS
			}
		}
		remove_from_array = {
			array = global.majors 
			value = THIS
		}
	}
}

# for one country
update_global_arrays = {
	update_majors_array = yes
}

remove_from_global_arrays_on_annex = {
	if = {

				value = THIS
			}
		}
		remove_from_array = {
			array = global.majors 
			value = THIS
		}
	}
}
