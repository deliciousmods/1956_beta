DANUBIC_create_danubic_array = {
	every_country = {
		limit = {
			DANUBIC_is_danubic = yes 
		}
		add_to_array = {
			array = PREV.DANUBIC_array 	
		}
	}
	for_each_scope_loop = {
		array = DANUBIC_array
		log = "[THIS.GetName] is part of Danubia"
	}
}
