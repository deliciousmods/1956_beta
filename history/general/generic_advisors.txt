# Political advisors
every_possible_country = {
	limit = {
		NOT = { OR = { tag = GER tag = ENG tag = SOV original_tag = FRA tag = ITA tag = JAP tag = USA tag = AST tag = RAJ tag = CAN tag = SAF tag = CZE tag = ROM tag = CHI tag = MAN tag = MEX tag = HOL original_tag = SPR original_tag = POL tag = FIN tag = VEN tag = DEN } }
		# avoid generic advisors in temporary countries
		is_dynamic_country = no
		NOT = {
			has_allowed_idea_with_traits = { idea = communist_revolutionary characters = yes }
		}
	}
	generate_character = {
		token_base = generic_communist_revolutionary # token will be TAG_token_base
		advisor = {
			slot = political_advisor
			traits = { communist_revolutionary }
			visible = {
				if = {
					limit = { has_focus_tree = turkish_focus }
					NOT = { tag = TUR }
				}
				if = {
					limit = { has_focus_tree = greek_focus }
					NOT = { tag = GRE }
				}
			}
			available = {
				if = {
					limit = {
						original_tag = NZL
						has_global_flag = NZL_tfv_focus_tree

					}
					has_completed_focus = NZL_the_lee_affair
				}
				if = {
					limit = {
						tag = NZL
						NOT = {
							has_global_flag = NZL_tfv_focus_tree
						}
					}
					has_completed_focus = NZL_emerge_from_the_caves
				}
				if = {
					limit = { has_dlc = "Man the Guns" }
					NOT = { has_autonomy_state = autonomy_supervised_state }
				}
			}

			traits = { communist_revolutionary }
			ai_will_do = {
				factor = 0
				modifier = {
					has_idea = internationalism
					NOT = {
						OR = {
							AND = {
								is_puppet = yes
								NOT = {
									has_government = communism
								}
							}
							communism > 0.7
							communism < 0.15
						}
					}
					add = 20
				}
			}

		}
		portraits = {
			army = { small = GFX_idea_generic_communist_revolutionary }
		}
	}
}
