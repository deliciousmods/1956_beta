country_event = { #Fate of Hawaii - what to do with them?
	id = r56_usa.6
	title = r56_usa.6.t
	desc = r56_usa.6.desc
	picture = GFX_report_event_partition_of_india

	is_triggered_only = yes
	option = { 	
		name = r56_usa.6.a #FREEEDOM!
		ai_chance = {
			factor = 25
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		if = {
			limit = {
				any_state = {
					is_core_of = HAW
					is_owned_by = ROOT
				}
			}
			effect_tooltip ={
				release = HAW
			}
			hidden_effect = {
				if = {
					limit = { HAW = { exists = no } }
					release = HAW
					HAW = { save_event_target_as = released_nation_target }
				}
				else = {
					create_dynamic_country = {
						original_tag = HAW
						
						save_event_target_as = released_nation_target
						ROOT = {
							every_owned_state = {
								limit = { is_core_of = HAW }
								ROOT = { transfer_state = HAW }
							}
						}
					}
				}
				diplomatic_relation = {
					country = HAW
					relation = docking_rights
					active = yes
				}
				ROOT = {give_guarantee = HAW }
			}
		}
		USA_congress_medium_support = yes
	}

	option = { 	
		name = r56_usa.6.b #Some autonomy
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		add_political_power = 15
		USA_congress_small_support = yes
		effect_tooltip ={
			release_puppet = HAW
		}
			
		hidden_effect = {
			if = {
				limit = { HAW = { exists = no } }
				release_puppet = HAW 
				set_autonomy = { target = HAW autonomy_state = autonomy_protectorate }
				HAW = { save_event_target_as = released_nation_target }
			}
			else = {
				create_dynamic_country = {
					original_tag = HAW
					
					save_event_target_as = released_nation_target
					ROOT = {
						every_owned_state = {
							limit = { is_core_of = HAW }
							ROOT = { transfer_state = HAW }
						}
					}
					ROOT = {
						puppet = HAW

					}
				}
			}
		}
