			ENG = { transfer_state = 695 }
		}

		trigger = {
			is_in_faction_with = USA

	id = mtg_netherlands.80
	title = mtg_netherlands.80.t
	desc = mtg_netherlands.80.d
	picture = GFX_report_event_generic_sign_treaty1

	is_triggered_only = yes
	
	# Accept
	option = {
		name = mtg_netherlands.80.a
		ai_chance = {
			base = 5
			modifier = {
				factor = 2
				has_government = fascism #we're interested in that region!
			}
			modifier = { #We are afraid of your enemies!
				factor = 0
				NOT = {
					has_government = fascism
				}
				HOL = {
					has_war_with = SOV
				}
			}
			modifier = { #Don't drag us into your wars!
				factor = 0
				has_war_with = SOV
				HOL = {
					any_enemy_country = {
						is_major = yes
						NOT = {
							has_war_with = JAP
						}
					}
				}
			}			
		}
		if = {
			limit = { #Create new alliance
				is_in_faction = no
				HOL = {
					is_in_faction = no
				}
			}
			HOL = { 
				create_faction = HOL_JAP_new_faction
				add_to_faction = JAP
			}
		}
		else_if = { #might break in rare scenarios
			limit = { #Japan in empty alliance
				all_allied_country = {
					OR = {
						is_subject = yes
						tag = SIA #we dont care about them lol
					}
				}
			}
			HOL = {
				if = {
					limit = {
						is_faction_leader = no
					}
					create_faction = HOL_JAP_new_faction
				}
				add_to_faction = JAP
			}
		}
		else = { #Join Japan's alliance instead
			add_to_faction = HOL
		}
	}
	# Decline
	option = {
		name = mtg_netherlands.80.b
		ai_chance = {
			base = 5
		}
	}
