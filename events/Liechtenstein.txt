# Death of Franz I
country_event = {
    id = liechtenstein.1
    picture = GFX_report_event_europe_funeral
	title = liechtenstein.1.t
	desc = liechtenstein.1.d
    fire_only_once = yes

    trigger = {
        original_tag = LIE
        date > 1938.7.25
        has_country_leader = {
            character = LIE_franz_i
        }
    }

    mean_time_to_happen = {
        days = 20
    }

    immediate = {
        hidden_effect = {
            retire_character = LIE_franz_i
        }
    }

    option = {
        name = liechtenstein.1.a
        ai_chance = {
			base = 10
        }
        if = {
            limit = {
                has_idea = LIE_franz_i
            }
            swap_ideas = {
                remove_idea = LIE_franz_i
                add_idea = LIE_franz_ii
            }
        }
        add_country_leader_role = {
            character = LIE_franz_ii
            promote_leader = yes
            country_leader = {
                ideology = absolute_monarchism
                expire = "1965.1.1.1"
                traits = { LIE_franz_joseph_ii_prince_of_liechtenstein }
            }
        }
    }

}
