add_namespace = liechtenstein

# Death of Franz I
country_event = {
    id = liechtenstein.1
    picture = GFX_report_event_europe_funeral
	title = liechtenstein.1.t
	desc = liechtenstein.1.d
    fire_only_once = yes

    trigger = {
        date > 1938.7.25
        has_country_leader = {
            character = LIE_franz_i
        }
    }

    mean_time_to_happen = {
        days = 20
    }

    immediate = {
        hidden_effect = {
            retire_character = LIE_franz_i
        }
    }

    option = {
        name = liechtenstein.1.a
        ai_chance = {
			base = 10
        }
        retire_character = LIE_franz_i
        add_country_leader_role = {
            character = LIE_franz_ii
            promote_leader = yes
            country_leader = {
                ideology = absolute_monarchism
                expire = "1965.1.1.1"
            }
        }
    }

}

# Liechtenstein suggests further cooperation with Switzerland

    id = liechtenstein.2
    picture = GFX_report_event_gre_diplomacy
	title = liechtenstein.2.t
	desc = liechtenstein.2.d
    fire_only_once = yes

    is_triggered_only = yes

    option = { # Yes
        name = liechtenstein.2.a
        ai_chance = {
			factor = 10
            modifier = {
                factor = 100
                is_historical_focus_on = yes
            }
        }
        if = {
            limit = {
                NOT = {
                    has_non_aggression_pact_with = LIE
                }
            }
            diplomatic_relation = {
                country = LIE
                relation = non_aggression_pact
                active = yes
            }
        }
        if = {
            limit = {
                NOT = {
                    has_guaranteed = LIE
                }
            }
            diplomatic_relation = {
                country = LIE
                relation = guarantee
                active = yes
            }
        }
        if = {
            limit = {
                NOT = {
                    has_military_access_to = LIE
                }
            }
            LIE = {
                diplomatic_relation = {
                    country = SWI
                    relation = military_access
                    active = yes
                }
            }
        }
        LIE = {
            country_event = { id = liechtenstein.3 hours = 6 }
        }
    }
    option = {
        name = liechtenstein.2.b
        ai_chance = {
			factor = 2
        }
        LIE = {
            country_event = { id = liechtenstein.4 hours = 6 }
        }

    }
}
