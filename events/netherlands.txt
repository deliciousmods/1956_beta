#fighter competition - light fighter competition winner gets a choice whether to permit license production
country_event = {
	id = netherlands.7
	title = netherlands.7.t
	desc = netherlands.7.desc 
	picture = GFX_report_event_fighters
	
	is_triggered_only = yes

	option = {# certainly
		name = netherlands.7.a
		ai_chance = {
			base = 90
			modifier = {
				has_war = yes
				factor = 0.5
			}
		}
		HOL = { country_event = netherlands.9 }
		add_relation_modifier = {
			target = HOL
			modifier = free_license # name is from static_modifiers.txt
		}
		if = {
			limit = {
				has_tech = fighter1
				NOT = { has_tech = fighter2 }
			}
			create_production_license = {
				target = HOL 
				equipment = {
					type = fighter_equipment_1
					
				}
				cost_factor = 0
			}
		}
		if = {
			limit = {
				has_tech = fighter2
				NOT = { has_tech = fighter3 }
			}
			create_production_license = {
				target = HOL 
				equipment = {
					type = fighter_equipment_2
				}
				cost_factor = 0
			}
		}
		if = {
			limit = {
				has_tech = fighter3
			}
			create_production_license = {
				target = HOL 
				equipment = {
