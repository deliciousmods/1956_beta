#fighter competition - light fighter competition winner gets a choice whether to permit license production
country_event = {
	id = belgium.104
	title = belgium.104.t
	desc = belgium.104.desc 
	picture = GFX_report_event_fighters
	
	is_triggered_only = yes

	option = {# certainly
		name = belgium.104.a
		ai_chance = {
			factor = 90
			modifier = {
				has_war = yes
				factor = 0.5
			}
		}
		BEL = { country_event = belgium.106 }
		add_relation_modifier = {
			target = BEL
			modifier = free_license # name is from static_modifiers.txt
		}
		if = {
			limit = {
				has_tech = fighter1
				NOT = { has_tech = fighter2 }
			}
			create_production_license = {
				target = BEL 
				equipment = {
					type = fighter_equipment_1
					
				}
				cost_factor = 0
			}
		}
		if = {
			limit = {
				has_tech = fighter2
				NOT = { has_tech = fighter3 }
			}
			create_production_license = {
				target = BEL 
				equipment = {
					type = fighter_equipment_2
				}
				cost_factor = 0
			}
		}
		if = {
			limit = {
				has_tech = fighter3
			}
			create_production_license = {
				target = BEL 
				equipment = {
					type = fighter_equipment_3
				}
				cost_factor = 0
			}
		}
	}

	option = {# no
		name = belgium.104.b
		ai_chance = {
			factor = 10
			modifier = {
				is_in_faction_with = BEL
				factor = 0
			}
		}
		if = {
			limit = { tag = GER }
			BEL = { set_country_flag = GER_fighter_refused }
		}
		if = {
			limit = { tag = ITA }
			BEL = { set_country_flag = ITA_fighter_refused }
		}
		if = {
			limit = { tag = FRA }
			BEL = { set_country_flag = FRA_fighter_refused }
		}
		if = {
			limit = { tag = ENG }
			BEL = { set_country_flag = ENG_fighter_refused }
		}
		if = {
			limit = { tag = USA }
			BEL = { set_country_flag = USA_fighter_refused }
		}
		BEL = { country_event = belgium.102 }
		reverse_add_opinion_modifier = { target = BEL modifier = refused_help }
	}
}
