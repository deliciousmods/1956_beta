country_event = {
	id = nepal.1
	title = nepal.1.t
	desc = nepal.1.d
	picture = GFX_report_event_polish_politics

	trigger = {
		original_tag = NEP
		is_subject = no
		is_in_faction = no
		has_capitulated = no
		OR = {
			has_government = neutrality
			has_government = democratic
		}
		ENG = {
			any_country = {
				OR = {
					tag = GER
					tag = ITA
					tag = JAP
					tag = USA
					tag = SOV
					tag = CHI
					tag = FRA
				}
				has_offensive_war_with = ENG
			}
			OR = {
				has_government = democratic
				has_government = neutrality
			}
			is_faction_leader = yes
			has_capitulated = no
			NOT = {
				has_war_with = NEP
				is_in_faction_with = NEP
			}
		}
		NEP = {
			any_neighbor_country = {
				OR = {
					tag = ENG
					is_subject_of = ENG
					is_in_faction_with = ENG
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 20
	}

	fire_only_once = yes

	option = { #We must aid our allies. Immediately declare war!
		name = nepal.1.a
		ENG = {
			add_opinion_modifier = {
				target = NEP
				modifier = large_increase
			}
			add_ai_strategy = {
				type = alliance
				id = ROOT
				value = 100
			}
			add_ai_strategy = {
				type = befriend
				id = ROOT
				value = 100
			}
			country_event = { id = generic.2 hours = 18 }
		}
		add_ai_strategy = {
			type = alliance
			id = ENG
			value = 100
		}
		add_ai_strategy = {
			type = befriend
			id = ENG
			value = 100
		}
		ai_chance = {
			base = 75
		}
	}
	option = { #We must preserve our national interests...
		name = nepal.1.b
		if = {
			limit = {
				is_subject = no
				is_in_faction = no
				has_war = no
				has_government = neutrality
			}
			add_ideas = neutrality_idea
		}
		ai_chance = {
			base = 25
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
	}	
}
