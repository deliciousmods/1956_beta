	id = stability.3
	title = stability.3.t	
	desc = stability.3.desc	#
	picture = GFX_					#

		NOT = { 
			has_country_flag = crisis_in_progress
		}
		NOT = { # Don't fire stability events if the country is not mobilising at all

			has_idea = volunteer_only
			has_idea = undisturbed_isolation
			has_idea = isolation
			has_idea = civilian_economy
		}
		NOT = { has_completed_focus = SPR_all_must_bear_the_torch }	#regional defense council of spain with stability drop
		NOT = { has_completed_focus = HON_free_our_markets }		#ancap honduras with constant stability drop
		NOT = { has_completed_focus = MON_oust_genden }				#ancom mongolia with constant stability drop
	}

	mean_time_to_happen = { 
		days = 360 
		modifier = {
			factor = 2.0
			has_idea = disarmed_nation
		}
		modifier = {
			factor = 2.0
			has_idea = political_loyalty_spirit
		}		
		modifier = {
			factor = 1.5
			has_idea = volunteer_only
		}
		modifier = {
			factor = 2.5
			has_idea = undisturbed_isolation
		}
		modifier = {
			factor = 2.0
			has_idea = isolation
		}
		modifier = {
			factor = 1.5
			has_idea = civilian_economy
		}
		modifier = {
			factor = 2
			NOT = { has_government = democratic }
		}
		modifier = {
			factor = 0.75
			OR = {
				has_stability < 0.4
				has_war_support < 0.4
			}
		}
		modifier = {
			factor = 0.75
			OR = {
				has_stability < 0.2
				has_war_support < 0.2
			}
		}
	}

	immediate = { 
		random_list = {
			20 = { 
				modifier = {
					factor = 0
					has_country_flag = draft_dodging_resolved
				}
				modifier = {
					factor = 0
					has_idea = disarmed_nation
				}
				modifier = {
					factor = 0
					has_idea = volunteer_only
				}
				modifier = {
					factor = 0
					has_idea = MPL_collapse_idea
				}

					factor = 0.1
					has_war_support > 0.5
				}

			} #draft dodging

				modifier = {
					factor = 0
					has_country_flag = strikes_resolved
				}

					factor = 0
