#Stalin pass away
country_event = {
	id = prc.5
	title = prc.5.t
	desc = prc.5.d
	picture = GFX_report_event_stalin_propaganda

	trigger = {
		OR = {
			AND = {
				PRC = {
					is_ai = yes
				}
				date > 1952.3.1
			}
			AND = {
				date > 1944.10.15
				has_war = no
				PRC = {
					is_ai = no
				}
			}
			AND = {
				date > 1947.5.1
				has_war = yes
				PRC = {
					is_ai = no
				}
			}
		}
		original_tag = SOV
		NOT = { has_country_flag = counter_purge }
		has_government = communism
		has_country_leader = {				#making sure Stalin is dying!
			character = SOV_iosif_stalin
			ruling_only = yes
		}
	}

	mean_time_to_happen = { days = 365 }	#was 120, why though?!

	fire_only_once = yes

	immediate = {
		hidden_effect = {
			kill_country_leader = yes
			news_event = { id = kns.4 hours = 5 }
			create_country_leader = {
				name = "Georgy Malenkov"
				desc = " "
				picture = "gfx/leaders/SOV/r56_portrait_SOV_Georgy_Malenkov.dds"
				expire = "1965.1.1"
				ideology = marxism
				traits = {
					sov_power_truggle
				}
			}
			set_global_flag = stalin_dead
		}
	}

	option = {
		name = prc.5.a
		custom_effect_tooltip = sov_power_struggle_begins
	}
}

country_event = {
	id = prc.14
	title = prc.14.t
	desc = prc.14.d
	picture = GFX_report_event_second_vienna_award_pact

	trigger = {
		tag = PRC
		focus_progress = {
			progress > 0.1
			focus = PRC_economic_reform
		}
	}
	mean_time_to_happen = { days = 2 }

	fire_only_once = yes

	option = {
		name = prc.14.a
		random_owned_controlled_state = {
			limit = {
				is_core_of = ROOT
				free_building_slots = {
					building = industrial_complex
					size > 2
					include_locked = yes
				}
				free_building_slots = {
					building = infrastructure
					size > 3
				}
			}
			add_extra_state_shared_building_slots = 3
			add_building_construction = {
				type = industrial_complex
				level = 3
				instant_build = yes
			}
			add_building_construction = {
				type = infrastructure
				level = 4
				instant_build = yes
			}
		}
		custom_effect_tooltip = prc_more_trade_deals_tt
		every_other_country = {
			add_opinion_modifier = {
				target = PRC
				modifier = PRC_swcc_2
			}
		}
		if = {
			limit = {
				has_idea = PRC_socialism_with_chinese_characteristics
			}
			swap_ideas = {
				remove_idea = PRC_socialism_with_chinese_characteristics
				add_idea = PRC_socialism_with_chinese_characteristics_2
			}
		}
	}
}
