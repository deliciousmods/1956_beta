country_event = {
	id = mongol.20
	title = mongol.20.t
	desc = mongol.20.d
	picture = GFX_report_event_MON_cavalry_raid_good

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			save_event_target_as = our_country
			FROM = {
				save_event_target_as = our_state
			}
			set_variable = { num_guns_from_small_raid = 0 }
			randomize_variable = {
				var = num_guns_from_small_raid
				distribution = uniform
				min = 150
				max = 350
			}
			if = {
				limit = {
					FROM.FROM = {
						NOT = {
							has_country_flag = MON_raid_ignored
						}
					}
				}
				FROM.FROM.FROM = {	#just because they will get it removed in event chain
					clr_state_flag = border_incident_active
					save_event_target_as = their_state
				}
				FROM = {
					clr_state_flag = border_incident_active
				}
			}
			FROM.FROM = { 
				save_event_target_as = their_country
				if = {
					limit = {
						has_country_flag = MON_raid_ignored
					}
					clr_country_flag = MON_raid_ignored
				}
			}
			remove_targeted_decision = { target = event_target:their_country decision = MON_small_raid_cancelled }
		}
	}
	
	#Bring in the trophies and study them!
	option = {
		name = mongol.20.a
		add_equipment_to_stockpile = { 
			type = infantry_equipment 
			amount = num_guns_from_small_raid
			producer = FROM.FROM
		}
		if = {
			limit = {
				NOT = {
					has_global_flag = MON_had_won_the_first_small_raid
				}
			}
			add_tech_bonus = {
				name = infantry_weapons_bonus
				bonus = 0.25
				uses = 1
				category = infantry_weapons
			}
			set_global_flag = MON_had_won_the_first_small_raid
		}
	}
}
