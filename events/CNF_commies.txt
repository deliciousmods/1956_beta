	id = prc.5
	title = prc.5.t
	desc = prc.5.d
	picture = GFX_report_event_stalin_propaganda

	trigger = {
		OR = {
			AND = {
				PRC = {
					is_ai = yes
				}
				date > 1953.3.1
			}
			AND = {
				date > 1945.10.15
				has_war = no
				PRC = {
					is_ai = no
				}
			}
			AND = {
				date > 1948.5.1
				has_war = yes
				PRC = {
					is_ai = no
				}
			}
		}
		tag = SOV
		NOT = { has_country_flag = counter_purge }
		has_government = communism
	}

	mean_time_to_happen = { days = 1 }

	fire_only_once = yes

	immediate = {
		hidden_effect = {
			kill_country_leader = yes

			country_event = { id = prc.31 days = 600 }
			create_country_leader = {
				name = "Georgy Malenkov"
				desc = "POLITICS_PS_DESC"
				picture = "gfx/leaders/SOV/Portrait_Soviet_Generic_1.dds"
