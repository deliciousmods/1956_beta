add_namespace = samoa


country_event = { # Elections!
	id = samoa.1
	title = samoa.1.t	
	desc = samoa.1.desc	
	picture = GFX_report_event_election_vote
	
	is_triggered_only = yes
	
	fire_only_once = yes
	
	option = {  # Election Results
		name = samoa.1.a
		if = {
		    limit = {
			    has_completed_focus = SAM_with_olaf_nelson
			}
			country_event = samoa.2
		}
		
		else_if = {
		    limit = {
			    has_completed_focus = SAM_with_the_empire
			}
			country_event = samoa.3
		}
	}
}
