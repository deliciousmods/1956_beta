country_event = {
	id = SAF_event.1
	title = SAF_event.1.t
	desc = SAF_event.1.desc
	picture = GFX_report_event_saf_soldiers

	fire_only_once = yes

	trigger = {
		is_in_faction_with = ENG
		has_war_together_with = ENG
		has_government = neutrality
		has_country_leader = { ruling_only = yes character = SAF_j_b_m_hertzog }
	}
	mean_time_to_happen = {
		days = 2
	}

	option = {
		name = SAF_event.1.a
		add_political_power = -20
		add_stability = -0.05
		add_war_support = 0.05
		retire_country_leader = yes
		set_politics = { ruling_party = democratic elections_allowed = yes }
		add_popularity = {
			ideology = democratic
			popularity = 0.35
		}
		add_popularity = {
			ideology = fascism
			popularity = 0.05
		}				
		create_field_marshal = {
			name = "Jan Smuts"
			picture = "Portrait_SouthAfrica_Jan_Smuts.dds"
			traits = { politically_connected air_force_ties organizer }
			skill = 3
			id = 5602
			attack_skill = 2
			defense_skill = 2
			planning_skill = 2
			logistics_skill = 4
		}

		if = {
			limit = {
				ROOT = { is_subject_of = ENG }
			}
			add_ideas = SAF_ossewabrandwag_sabotages
		}
	}
}

	is_triggered_only = yes

	option = {
		name = SAF_event.4.a
		set_politics = { ruling_party = fascism elections_allowed = no }
		if = {
			limit ={
				is_in_faction = yes
			}
			faction_leader = {
				remove_from_faction = ROOT
			}
		}
		add_ideas = ideological_fanaticism_focus
	set_rule = {
		can_create_factions = yes
	}
		OVERLORD = {
			set_autonomy = {
				target = SAF
				autonomy_state = autonomy_free
			}
		}
		if = {
			limit = {
				GER = {
					has_government = fascism
				}
			}
			GER = {
				add_opinion_modifier = {
					target = ROOT
					modifier = small_increase
				}
			}
				ROOT = {
					add_opinion_modifier = {
						target = GER
						modifier = small_increase
					}
			}			
		}

		set_party_name = {
			ideology = neutrality
			long_name = SAF_anc_long
			name = SAF_anc_long
		}
		add_popularity = {
				ideology = neutrality
				popularity = 0.05
		}
	}
}
