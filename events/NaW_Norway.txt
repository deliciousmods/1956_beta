	is_triggered_only = yes
	
	option = {
		name = naw_norway.1.a ### Accept joining the Norwegian Three Brothers faction.
		ai_chance = { base = 90 }
		FROM = {
			country_event = naw_norway.2
			add_to_faction = ROOT
		}
		add_ai_strategy = {
			type = alliance
			id = "NOR"
			value = 250
		}
	}

	option = {
		name = naw_norway.1.b ### Reject joining the Norwegian Three Brothers faction.
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				has_opinion = {
					target = FROM
					value > 30
				}
			}
			modifier = {
				factor = 5
				has_opinion = {
					target = FROM
					value < -20
				}
			}
		}
		FROM = { country_event = naw_norway.3 }
	}
}

country_event = {
	id = naw_norway.7
	title = naw_norway.7.t
	desc = naw_norway.7.d
	picture = GFX_report_event_physics_lab_01
	
	is_triggered_only = yes
	
	option = {
		name = naw_norway.7.a ### Sounds good.
		trigger = {
			NOT = { has_dlc = "Together for Victory" }
		}
		ai_chance = { base = 90 }
		add_political_power = -10
		add_tech_bonus = {
			name = NOR_naval_technology_bonus
			bonus = 1.0
			uses = 1
			category = naval_equipment
		}
		FROM = {
			add_tech_bonus = {
				name = industrial_bonus
				bonus = 1.0
				uses = 1
				category = industry
			}
			add_tech_bonus = {
				name = NOR_naval_technology_bonus
				bonus = 1.0
				uses = 1
				category = naval_equipment
			}
			add_tech_bonus = {
				name = GER_long_range_escorts
				bonus = 1.0
				uses = 1
				category = cat_heavy_fighter
			}
			country_event = { days = 1 id = naw_norway.8 }
		}
	}
	
	option = {
		name = naw_norway.7.b ### Tech sharing.
		trigger = { has_dlc = "Together for Victory" }
		ai_chance = { base = 90 }
		add_political_power = -10
		if = {
			limit = {
				FROM = {
					NOT = { is_in_tech_sharing_group = NOR_nordic_research }
				}
			}
			FROM = { add_to_tech_sharing_group = NOR_nordic_research }
			every_other_country = {
				limit = {
					is_subject_of = FROM
					NOT = { is_in_tech_sharing_group = NOR_nordic_research }
				}
				add_to_tech_sharing_group = NOR_nordic_research
			}
		}
		if = {
			limit = {
				OR = {
					is_subject_of = FROM
					is_in_faction_with = FROM
				}
			}
			add_to_tech_sharing_group = NOR_nordic_research
		}
		FROM = {
			country_event = { days = 1 id = naw_norway.8 }
		}
	}
	
	option = {
		name = naw_norway.7.c ### No way.
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				has_opinion = {
					target = FROM
					value > 90
				}
			}
		}
		FROM = {
			country_event = { days = 1 id = naw_norway.9 }
		}
	}
}
