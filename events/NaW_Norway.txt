			NOT = { has_dlc = "Together for Victory" }
		}
		ai_chance = { base = 90 }
		add_political_power = -10
		add_tech_bonus = {
			name = NOR_naval_technology_bonus
			bonus = 1.0
			uses = 1
			category = naval_equipment
		}
		NOR = {
			add_tech_bonus = {
				name = industrial_bonus
				bonus = 1.0
				uses = 1
				category = industry
			}
			add_tech_bonus = {
				name = NOR_naval_technology_bonus
				bonus = 1.0
				uses = 1
				category = naval_equipment
			}
			add_tech_bonus = {
				name = GER_long_range_escorts
				bonus = 1.0
				uses = 1
				category = cat_heavy_fighter
			}
			country_event = { days = 1 id = naw_norway.8 }
		}
	}
	
	option = {
		name = naw_norway.7.b	#tech_sharing
		trigger = { has_dlc = "Together for Victory" }
		ai_chance = { base = 90 }
		add_political_power = -10
		if = {
			limit = {
				NOR = {
					OR = {
						is_in_faction_with = DEN
						is_in_faction_with = SWE
					}
					num_tech_sharing_groups < 1
				}
			}
			NOR = { add_to_tech_sharing_group = NOR_nordic_research }
		}
		if = {
			limit = {
				tag = DEN
				is_in_faction_with = NOR
			}
			DEN = { add_to_tech_sharing_group = NOR_nordic_research }
		}
		if = {
			limit = {
				tag = SWE
				is_in_faction_with = NOR
			}
			SWE = { add_to_tech_sharing_group = NOR_nordic_research }
		}

		}
