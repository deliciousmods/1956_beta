# The Fall of London (USA)
news_event = {
	id = cityfall.1
	title = cityfall.1.t
	desc = cityfall.1.d
	picture = GFX_news_event_fall_of_london
	
	major = yes
	
	trigger = {
		126 = { is_controlled_by = USA }
		126 = { is_owned_by = ENG }
		USA = { has_war_with = ENG }
		NOT = { has_global_flag = fall_of_london_usa }
	}
	
	immediate = {
		set_global_flag = fall_of_london_usa
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	option = {
		name = cityfall.1.a
		trigger = {
			NOT = {
				TAG = USA
				TAG = ENG
			}
		}
	}
	option = {
		name = cityfall.1.b
		trigger = { TAG = USA }
	}
	option = {
		name = cityfall.1.c
		trigger = { TAG = ENG }
	}
}
