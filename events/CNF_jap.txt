#kill pgr and rgr
country_event = {
	id = jap.108
	hidden = yes
	
	fire_only_once = yes
	
	trigger = {
		original_tag = CHI
		is_puppet_of = JAP
		OR = {
			PGR = {
				exists = yes
				is_puppet_of = JAP
			}
			RGR = {
				exists = yes
				is_puppet_of = JAP
			}
		}
	}
	
	mean_time_to_happen = { days = 0 }
	
	immediate = {
		if = {
			limit = {
				PGR = {
					is_puppet_of = JAP
					exists = yes
				}
			}
			annex_country = {
				target = PGR
				transfer_troops = yes
			}
		}
		if = {
			limit = {
				RGR = {
					is_puppet_of = JAP
					exists = yes
				}
			}
			annex_country = {
				target = RGR
				transfer_troops = yes
			}
		}
		if = {
			limit = {
				owns_state = 613
			}
			set_capital = 613
		}
	}
	
	option = {
	}
}
