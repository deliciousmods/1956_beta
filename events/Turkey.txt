add_namespace = turkey

# President dies
country_event = {
	id = turkey.1
	title = turkey.1.t
	desc = turkey.1.d
	picture = GFX_report_event_generic_funeral

	fire_only_once = yes
	
	trigger = {
		original_tag = TUR
		#has_government = neutrality
		date > 1938.11.1
		OR = {
			NOT = { has_dlc = "Battle for the Bosporus" }
			date > 1940.2.1 #creates a backup way to kill Ataturk
		}
		has_country_leader = {
			character = TUR_mustafa_kemal_ataturk
			ruling_only = yes
		}
		
	}	

	mean_time_to_happen = {
		days = 30
		modifier = {
			factor = 0.5
			date > 1938.12.1
		}
		modifier = {
			factor = 0.2
			date > 1939.1.1
		}
	}

	option = {
		name = turkey.1.a		#Ismet Inonu
		hidden_effect = {
			retire_character = TUR_mustafa_kemal_ataturk
		}
		add_stability = -0.1	#Ataturk doesn't give a stability buff in the road version
		promote_character = TUR_ismet_inonu
		ai_chance = {
			base = 10
			modifier = {
				is_historical_focus_on = yes
				factor = 100
			}
		}
	}
	
	option = {
		name = turkey.1.b		#Fevzi Cakmak
		trigger = {
			has_completed_focus = TUR_continue_the_moderate_course
		}
		hidden_effect = {
			retire_character = TUR_mustafa_kemal_ataturk
		}
		add_stability = -0.1	#Ataturk doesn't give a stability buff in the road version
		add_country_leader_role = {
			character = TUR_fevzi_cakmak
			country_leader = {
				expire = "1965.1.1"
				ideology = despotism
				traits = {
					maresal
				}
			}
			promote_leader = yes
		}
		ai_chance = {
			base = 10
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
	}
}
