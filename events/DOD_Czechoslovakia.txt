	id = czech.35
	
	trigger = {
		original_tag = CZE
		NOT = {
			owns_state = 70
			owns_state = 71
			owns_state = 73
			owns_state = 664
		}
		NOT = { has_war_with = SLO }
		NOT = { has_global_flag = CZE_became_czech }
	}
	
	mean_time_to_happen = { days = 2 }

	hidden = yes

	immediate = {
		set_cosmetic_tag = CZE_ONLY
		set_global_flag = CZE_became_czech
	}
}
