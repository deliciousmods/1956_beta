#England asks Germany for alliance
country_event = {
	id = englandmod.1
	title = englandmod.1.t
	desc = englandmod.1.d
	picture = GFX_report_event_fascist_gathering

	is_triggered_only = yes
		
	option = {
		name = englandmod.1.a
		ai_chance = { factor = 99 }
		ENG = { remove_from_faction = ENG }
		GER = { add_to_faction = ENG }
		ENG = {
			add_ai_strategy = {
				type = alliance
				id = "ENG"
				value = 200
			}
		}
		hidden_effect = {
			news_event = { hours = 6 id = newsmod.3 }
		}
	}
	option = {
		name = englandmod.1.b
		ai_chance = { factor = 1 }
	}
}

# England - Demand Ireland
country_event = {
	id = englandmod.5
	title = englandmod.5.t
	desc = englandmod.5.d
	picture = GFX_report_event_king_speech

	is_triggered_only = yes

	option = {
		name = englandmod.5.a
		ai_chance = {
			factor = 80
			modifier = {
				factor = 0
				any_other_country = {
					is_major = yes
					NOT = { tag = ENG }
					OR = {
						has_guaranteed = IRE
						is_in_faction_with = IRE
					}
				}
			}
		}
		ENG = { country_event = englandmod.6 }
	}

	option = {
		name = englandmod.5.b
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = ENG
						ENG = { is_faction_leader = yes }
					}
					AND = {
						is_in_faction = no 
						NOT = {
							any_other_country = {
								NOT = { tag = ENG }
								has_guaranteed = IRE
							} 
						}
					}
				}
			}
		}
		ENG = {
			effect_tooltip = {
				create_wargoal = {
					type = annex_everything
					target = IRE
				}
			}
			country_event = englandmod.7
		}
	}
}

# Remove Fascist Influence if a dominion becomes independent
country_event = {
	id = englandmod.23
	title = englandmod.23.t
	desc = englandmod.23.d
	trigger = {
		NOT = { is_subject_of = ENG 
		has_country_flag = kept_dominion_fascist_influence
		}
		has_idea = dominion_fascist_influence
	}
	fire_only_once = yes
	option = {
		name = englandmod.23.a
		ai_chance = { factor = 90 }
			remove_ideas = dominion_fascist_influence
		}
	option = {
		name = englandmod.23.b
		ai_chance = { factor = 10 }
		set_country_flag = kept_dominion_fascist_influence	
		}
	mean_time_to_happen = {
		days = 1
	}
}
