						}

							add_ai_strategy = {

							}
						}
					}
				}
			}
		}
		if = {
			limit = {
				tag = AST
			}
			FROM = { set_autonomy = { target = AST autonomy_state = autonomy_free } }
			if = {
				limit = {
					CAN = { NOT = { has_country_flag = ENG_abdication_crisis_dominion_will_leave } }
				}
				create_faction = ENG_commonwealth_of_nations

					limit = {
						SAF = { has_country_flag = ENG_abdication_crisis_dominion_will_leave }
					}
					add_to_faction = SAF
				}

					limit = {

					add_to_faction = NZL

							NOT = {
								has_government = fascism
								has_government = communism
							}

					}
					add_to_faction = OMA
				}
				hidden_effect = {

						limit = {
							NOT = { tag = SAF }
							NOT = { tag = NZL }
							NOT = { tag = OMA }

							add_ai_strategy = {
								type = alliance
								id = PREV
								value = -200
							}
						}
					}

			}

					has_country_flag = ENG_abdication_crisis_dominion_will_leave

					}

	}

	# Reject

		name = mtg_britain.88.b

			factor = 50

				factor = 0.8

				factor = 0.8

			}
